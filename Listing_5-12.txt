module TopLevel (SW, LEDR);
 input [9:0] SW;
 output [9:0] LEDR;
 wire AB, AC, BC; // Internal connections for carry
 xor SumSig (SW[0], SW[1], SW[2], LEDR[0]); // Sum = A ^ B ^ C
 and (SW[0], SW[1], AB);
 and (SW[0], SW[2], AC);
 and (SW[1], SW[2], BC);
 or CarrySig (AB, AC, BC, LEDR[1]); // Carry = A & B | A & C | B & C
endmodule
